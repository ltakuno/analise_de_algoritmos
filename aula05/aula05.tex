\begin{slide}
  \stitle{Algoritmo Heap-Sort}  
  
  \textbf{Heap:} Estrutura de dados útil quando se quer remover o máximo de um conjunto sucessivas vezes.
\end{slide}
 
\begin{slide}
  \stitle{Algoritmo Heap-Sort}  
  
  Formalmente, um heap é uma árvore binária em que cada nó contém um elemento do conjunto e vale:
  \begin{enumerate}
  \item A árvore é completa até o penúltimo nível
  \item No último nível todos os nós estão o mais a esquerda possível
  \item Cada nó interno armazena um valor maior que seus filhos
  \end{enumerate}
\end{slide}

\begin{slide}
  \stitle{Algoritmo Heap-Sort}  
Exemplo: $S = \{20,15,17,-2,8,11,3,9,29,1\}$

\begin{slide}
  \stitle{Algoritmo Heap-Sort}  
Um heap pode ser armazenado em um vetor de tal forma que:

\itemtrig\ O pai do elemento da posição $i$ ($i>1$) sempre está na posição $\lfloor i/2 \rfloor$

\vspace{-1cm}\itemtrig\ E os filhos, nas posições $2i$ e $2i+1$ (se tais índices forem menores ou iguais a n).

Note que o valor máximo do conjunto representado pelo heap sempre está na raiz.
\end{slide}
 
\begin{slide}
  \stitle{Algoritmo Heap-Sort} 
  Algoritmos de manipulação do heap consome tempo proporcional à altura do heap. A altura do heap com n elementos: $\Theta(\log n)$
  
\begin{codebox}
\Procname{$\proc{Desce-Heap}(A, n, i)$}
\li  \While $(2i \leq n)$
\li  \Do   $f = 2i$ 
\li  \If $(f < n)$ $e$ $(A[f+1]>A[f])$
\li \Then $f = f + 1$
\End
\li  \If $(A[i]<A[f])$
\li \Then $A[i] \leftrightarrow A[f]$
\li       $i = f$
\li \Else
\li       $i = n + 1$
\End
\End
\End
\end{codebox}  
\end{slide}
 
\begin{slide}
  \stitle{Remoção do máximo heap} 
  
\begin{codebox}
\Procname{$\proc{Extrai-Maximo-Heap}(A, n)$}
\li  $max = A[1]$ 
\li  $A[1] = A[n]$ 
\li  $\proc{Desce-Heap}(A, n-1, 1)$
\li  \Return $max$
\End
\end{codebox}  
\end{slide}

 
\begin{slide}
  \stitle{Construção de um Heap} 
  Podemos utilizar o $\proc{Desce-Heap}()$
\begin{codebox}
\Procname{$\proc{Constroi-Heap}(A)$}
\li  \For $i$ $de$ $\lfloor n/2 \rfloor$ \To $1$ 
\li  \Do	$\proc{Desce-Heap}(A, n, i)$
\End
\end{codebox}  
  Custo do $\proc{Constroi-Heap}()$: $\Theta(n)$

Agora podemos definir o algoritmo $\proc{Heap-Sort}$

\end{slide}

\begin{slide}
  \stitle{Heap-Sort} 
  
\begin{codebox}
\Procname{$\proc{Heap-Sort}(A, n)$}
\li  $\proc{Constroi-Heap(A)}$ 
\li  \For $i = n$ \To $2$    \hspace{2.5cm} $\vartriangleright$ $O(n \log n)$
\li  \Do $A[1] \leftrightarrow A[i]$
\li  $\proc{Desce-Heap(A, i-1, 1)}$
\End
\end{codebox}  
\end{slide}

\begin{slide}
  \stitle{Complexidade do Heap-Sort} 
\textbf{Pior caso:} Vetor em ordem decrescente $O(n \log n)$

\textbf{Melhor caso:} Vetor em ordem crescente $O(n \log n)$

\textbf{Caso aleatório:} Vetor aleatório $O(n \log n)$

\textbf{Complexidade de expaço:} $\Theta(n)$
\end{slide}

\begin{slide}
  \stitle{Algoritmo Quick-Sort} 
  Estratégia de divisão e conquista
  
\begin{codebox}
\Procname{$\proc{Quick-Sort}(A, p, r)$}
\li  \If $p < r$
\li  \Then $q = \proc{particiona}(A, p, r)$  $\hspace{2cm}$  $\vartriangleright$ $p \leq q < r$
\li  $\proc{Quick-Sort}(A, p, q)$
\li  $\proc{Quick-Sort}(A, q+1, r)$
\End
\end{codebox}   
  
\end{slide}

\begin{slide}
  \stitle{Particionamento}
\vspace{1.0cm}
$\proc{particiona}(A, p, r)$
\vspace{-2.0cm}
\begin{tabbing}
~~~~ \= ~~~~\= ~~~~\=  ~~~~\= \hspace*{12cm} \= custo \\ 
1  \> $x = A[r]$ \>   \> \> \> $\Theta(1)$\\
2  \> $i = p-1$ \>   \> \> \> $\Theta(1)$\\
3  \> \For $j = p$ \To  $r-1$  \textbf{\blue faça} \>   \> \> \> $\Theta(n)$      \\
4  \> \>  \If $A[j] \leq x$ \textbf{\blue então}                    \> \> \> $\Theta(n)$      \\
5  \> \> \> $i = i+1$                 \> \> $\Theta(n)$ \\
6  \> \> \> $A[i] \leftrightarrow A[j]$                 \> \> $\Theta(n)$ \\
\End
7  \> $A[i+1] \leftrightarrow A[r]$             \> \> \> \> $\Theta(1)$ \\
8  \> \Return $i+1$           \>  \> \> \> $\Theta(1)$ \\
   \> \> \> \> \> Total: $\Theta(n)$                  
\end{tabbing}  
  Portanto, a complexidade do algoritmo é $\Theta(n)$
\end{slide}

\begin{slide}
  \stitle{Complexidade do Quick-Sort}
    $T(n)= \left\{  \begin{array}{ll} 
                \Theta(1) & \textrm{, se } n=0 \\
                \max \{ T(m) + T(n-m-1)\} + \Theta(n) & \textrm{, se } n>0 \textrm{ e } 0 \leq m < n
                \end{array} \right. $
                
  Resolvendo a recorrência temos, que:
  
  $T(n) = O(n^2)$
  
  Dica: utilize indução para verificar que a solução é de fato esta.
\end{slide}

\begin{slide}
  \stitle{Complexidade do Quick-Sort}
Pior caso: partições não balanceadas

	$\begin{array}{l}
	T(n) = T(n-1) + \Theta(n)\\
	T(n) = \Theta(n^2)	
\end{array}$

Melhor caso: partições balanceadas

	$\begin{array}{l}
	T(n) = 2T(n/2) + \Theta(n)\\
	T(n) = \Theta(n \log n)	
\end{array}$

Caso médio: 

	$T(n) = \Theta(n \log n)$
\end{slide}


\begin{slide}
  \stitle{Quick-Sort aleatorizado} 

\begin{codebox}
\Procname{$\proc{particiona-aleatorizado}(A, p, r)$}
\li  $i = \proc{aleatorio}(p, r)$
\li  $A[r] \leftrightarrow A[i]$
\li  $\Return$ $\proc{particiona}(A, p, r)$
\end{codebox} 

  
\begin{codebox}
\Procname{$\proc{Quick-Sort-Aleatorizado}(A, p, r)$}
\li  \If $p < r$
\li  \Then $q = \proc{particiona-aleatorizado}(A, p, r)$  
\li  $\proc{Quick-Sort-Aleatorizado}(A, p, q)$
\li  $\proc{Quick-Sort-Aleatorizado}(A, q+1, r)$
\End
\end{codebox}   
\end{slide}


\begin{slide}
  \stitle{Análise do Quick-Sort aleatorizado}   
  Recorrência para o tempo esperado $E(T(n))$:
  
    $T(n)= \left\{  \begin{array}{ll} 
                \Theta(1) & \textrm{, se } n=1 \\\\
                \displaystyle \frac{1}{n} \sum^{n-1}_{k=1} (T(k) + T(n-k)) + \Theta(n) & \textrm{, se } n>1  
                \end{array} \right. $
                
Podemos manipular o somatório de forma adequada:
                
  $$T(n) = \frac{2}{n} \displaystyle \sum^{n-1}_{k=1} T(k) + cn$$
  \end{slide}


\begin{slide}
  \stitle{Análise do Quick-Sort aleatorizado} 
  Podemos provar que $T(n) = O(n \log n)$. Suponha que $T(n) \leq a n \log n + b$, para $n \geq n_0$, com $a, b > 0$ constantes.

$$
\begin{array}{rl}
T(n) & = \displaystyle \frac{2}{n} \sum^{n-1}_{k=1} T(k) + cn \\
     & \leq \displaystyle \frac{2}{n} \sum^{n-1}_{k=1} (a k\log k + b) + cn \\
     & = \displaystyle \frac{2a}{n} \sum^{n-1}_{k=1} k\log k + \frac{2b}{n}(n-1) + cn

\end{array}
$$  
    \end{slide}


\begin{slide}
  \stitle{Análise do Quick-Sort aleatorizado} 
  Sabemos que $\displaystyle \sum^{n-1}_{k=1} k\log k \leq \displaystyle\frac{1}{2} n^2\log n - \frac{1}{8}n^2$
  
$$
\begin{array}{rl}
T(n) & = \displaystyle \frac{2a}{n} \sum^{n-1}_{k=1} k\log k + \frac{2b}{n}(n-1) + cn \\
     & \leq \displaystyle  \frac{2a}{n} (\frac{1}{2} n^2\log n - \frac{1}{8}n^2) + \frac{2b}{n}(n-1) + cn \\
     & \leq \displaystyle  an\log n - \frac{a}{4} n + 2b + cn \\
     & = \displaystyle  an\log n + b + ( cn + b - \frac{a}{4} n)\\
&\leq  \displaystyle  an\log n + b \\
\end{array}
$$  
  Para $\displaystyle \frac{a}{4}n \geq cn + b$
\end{slide}
