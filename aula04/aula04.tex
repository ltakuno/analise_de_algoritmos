\begin{slide}
  \stitle{Insertion-sort recursivo}
\begin{codebox}
\Procname{$\proc{Insertion-Sort}(A, n)$}
\li \If $n \geq 2$
\li \Then $\proc{Insertion-Sort}(A, n-1)$
\li     $chave = A[i]$
\li     $j = n-1$
\li     \While $(j > 0)$ $e$  $(A[j] > chave)$
\li  \Do $A[j+1] = A[j]$
\li      $j = j - 1$
\End
\li   $A[j+1] = chave$
\End
\end{codebox}
\end{slide}


\begin{slide}
  \stitle{Análise do Insertion-sort recursivo}
  Quantas comparações e quantas trocas o algoritmo executa no pior caso?
  
\end{slide}

\begin{slide}
  \stitle{Análise do Insertion-sort recursivo}
  Tanto o número de comparações quanto o de trocas é dado pela recorrência:
  
  $T(n)= \left\{  \begin{array}{ll} 
                0 & \textrm{, se } n=1 \\
                T(n-1) + n & \textrm{, se } n>1
                \end{array} \right. $
 
 $\Theta(n^2)$ comparações e trocas são executadas no pior caso.
  
\end{slide}

  \stitle{Algoritmo Selection-Sort \\ Iterativo}

\begin{slide}
  \stitle{Algoritmo Selection-Sort}
\vspace{1.0cm}
$\proc{Selection-Sort}(A, n)$
\vspace{-2.0cm}
\begin{tabbing}
~~~~ \= ~~~~\= ~~~~\= ~~~~\= ~~~~~~~\= \hspace*{12cm}                               \= custo \\
1  \> \For $i = 1$ \To  $n-1$  \textbf{\blue faça}\> \>  \>  \\
2  \> \>  $\id{min} = i$ \>   \>   \\
3  \> \> \For $j = 	i + 1$ \To $n$ \textbf{\blue faça} \>   \>   \\
4  \> \> \> \If $A[j] < A[min]$ \textbf{\blue então} \> \> \\
5  \> \> \> \> $\id{min} = j$                            \>   \>   \\
6  \> \> $A[i] \leftrightarrow A[min]$  \> \\
\end{tabbing}  
\end{slide}

\begin{slide}
  \stitle{Algoritmo Selection-Sort}
\vspace{1.0cm}
$\proc{Selection-Sort}(A, n)$
\vspace{-2.0cm}
\begin{tabbing}
~~~~ \= ~~~~\= ~~~~\=  ~~~~\= \hspace*{12cm} \= custo \\ 
1  \> \For $i = 1$ \To  $n-1$  \textbf{\blue faça} \>   \> \> \> $\Theta(n)$      \\
2  \> \>  $\id{min} = i$                                \> \> \> $\Theta(n)$      \\
3  \> \> \For $j = 	i + 1$ \To $n$ \textbf{\blue faça}  \> \> \> $\Theta(n^2)$    \\
4  \> \> \> \If $A[j] < A[min]$ \textbf{\blue então}    \> \>    $\Theta(n^2)$    \\
5  \> \> \> \> $\id{min} = j$                           \>       $\Theta(n^2)$    \\
6  \> \> $A[i] \leftrightarrow A[min]$               \> \> \>   $\Theta(n^2)$    \\
   \> \> \> \> \> Total: $\Theta(n^2)$                  
\end{tabbing}  
  Portanto, a complexidade do algoritmo é $\Theta(n^2)$
\end{slide}

\begin{slide}
  \stitle{Análise do algoritmo Selection-Sort (iterativo)}

\itemtrig\ Complexidade de pior caso (elementos em ordem decrescente): $\Theta(n^2)$, com $\Theta(n^2)$ comparações e $\Theta(n)$ trocas.

\itemtrig\ Complexidade de melhor caso (elementos em ordem crescente): $\Theta(n^2)$ 

\itemtrig\ Complexidade de caso médio (vetor aleatório): $\Theta(n^2)$
\end{slide}

\begin{slide}
  \stitle{Análise do algoritmo Selection-Sort (iterativo)}

\itemtrig\ Complexidade de espaço: $\Theta(n)$, pois comparações e trocas são efetuadas diretamente no vetor.

Note que o Selection-Sort realiza mais comparações que o Insertion-Sort, mas menos trocas.    

\end{slide}

\begin{slide}
  \stitle{Análise do algoritmo Selection-Sort (iterativo)}

  
\textbf{Invariante:}

O vetor $A[1 \cdots i-1]$ está ordenado e $A[1 \cdots i-1] < A[i \cdots n]$

A partir desta informação valiosa demonstre a corretude do algoritmo Selection-Sort
\end{slide}

\begin{slide}
  \stitle{Análise do algoritmo Selection-Sort (recursivo)}
\end{slide}

\begin{slide}
  \stitle{Análise do algoritmo Selection-Sort (recursivo)}
\vspace{1.0cm}
$\proc{Selection-Sort}(A, i, n)$
\vspace{-2.0cm}
\begin{tabbing}
~~~~ \= ~~~~\= ~~~~\=  ~~~~\= \hspace*{12cm} \=  \\ 
1  \> \If $i < n$ \textbf{\blue então}  \\
2  \> \>  $\id{min} = i$                                \> \> \>     \\
3  \> \> \For $j = 	i + 1$ \To $n$ \textbf{\blue faça}  \> \> \>     \\
4  \> \> \> \If $A[j] < A[min]$ \textbf{\blue então} $\id{min} = j$   \> \>        \\
5  \> \> $t = A[min]$               \> \> \>        \\
6  \> \> $A[min] = A[i]$            \> \> \>        \\
7  \> \> $A[i] = t$            \> \> \>        \\
8  \> \> $\proc{Selection-Sort}(A, i + 1, n)$                
\end{tabbing}  
  
\end{slide}

\begin{slide}
  \stitle{Análise do algoritmo Selection-Sort (recursivo)}
  Quantas comparações e quantas trocas o algoritmo executa no pior caso?
  
  O número de comparações é dado pela recorrência:

  $T(n)= \left\{  \begin{array}{ll} 
                0 & \textrm{, se } n=1 \\
                T(n-1) + \Theta(n) & \textrm{, se } n>1
                \end{array} \right. $
   
   $\Theta(n^2)$ comparações são executadas no pior caso.
\end{slide}

\begin{slide}
  \stitle{Análise do algoritmo Selection-Sort (recursivo)}
  O número de trocas é dado pela recorrência:

  $T(n)= \left\{  \begin{array}{ll} 
                0 & \textrm{, se } n=1 \\
                T(n-1) + \Theta(1) & \textrm{, se } n>1
                \end{array} \right. $
  
   $\Theta(n)$ trocas são executadas no pior caso.
   
   
  $\proc{Insertion-Sort}$ e $\proc{Selection-Sort}$ têm a mesma complexidade assintótica, porém em situações em que a operação de troca é muito custosa, é preferível utilizar $\proc{Selection-Sort}$.
   
    
\end{slide}

\begin{slide}
  \stitle{Algoritmo Tim-Sort}
  Desenvolvido por Tim Peters em 2002 para a linguagem Python
  
  É um algoritmo de ordenação híbrido de Merge Sort e Insertion-Sort
  
  \itemtrig\ Complexidade de pior caso: $\Theta(n \log n)$

  \vspace{-1cm} \itemtrig\ Complexidade de melhor caso: $\Theta(n)$
  
  \vspace{-1cm} \itemtrig\ Complexidade de caso médio: $\Theta(n \log n)$
  
\end{slide}

\begin{slide}
  \stitle{Algoritmo Bogo-Sort}
  
  É um algoritmo de ordenação perversamente ineficiente
  
  É um algoritmo probabilístico por natureza
  
  Para ordenar os $n$ elementos distintos a complexidade esperada é $O(n \cdot n!)$
  
  Não é utilizada na prática, mas útil em disciplinas de análise de algoritmos
\end{slide}
