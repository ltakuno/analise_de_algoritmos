\begin{slide}
  \stitle{Problema do troco}  
Dado um valor $P$, e um conjunto $S$ de $n$ moedas, cada um com um valor $c_i$, precisamos determinar o número mínimo de moedas para obter a quantidade $P$.
\end{slide}

\begin{slide}
  \stitle{Problema do troco}  
  Perceba que temos três casos base:

  \itemtrig\ Se o valor do troco é 0, então temos 0 moedas como solução.
  
   \vspace{-1cm}\itemtrig\ Se o valor do troco é negativo ($< 0$), então não é possível dar o troco.
   
      \vspace{-1cm}\itemtrig\ Se o conjunto $S$ é vazio ($\{\}$ ou $\emptyset$), então não é possível dar o troco.
     
  
\end{slide}

\begin{slide}
  \stitle{Problema do troco}  
Definimos a função $m(i, Q)$ como o número mínimo de moedas necessárias para obter uma quantidade $Q$, usando os $i$ primeiros tipos de moedas ($1 \cdots i$).

A solução para o troco pode utilizar 0 ou mais moedas do tipo $i$, e assim:

$m(i, Q)= \left\{  \begin{array}{ll} 
                m(i-1, Q) & \textrm{, se não usa moeda tipo } i \\
                m(i-1, Q-k \cdot c_i)  & \textrm{, se usa } k \textrm{ moedas do tipo } i
                \end{array} \right. $
\end{slide}

\textbf{Mochila}  

Dados dois vetores $m[1 \cdots n]$ e $w[1 \cdots n]$, denotamos por $m \cdot w$ o produto escalar $m[1]w[1] + m[2]w[2] + \cdots + m[n]w[n]$.

Suponha um número inteiro não-negativo $W$ e vetores positivos $w[1 \cdots n]$ e $v[1 \cdots n]$.
\end{slide}

\begin{slide}
  \stitle{Problema da mochila}
  Uma mochila é qualquer vetor $m[1 \cdots n]$ tal que $m \cdot w \leq W$ e $0 \leq m[i] \leq 1$ para todo $i$.
  
  Denotamos por $W$ a capacidade da mochila. O \textbf{\blue valor} de uma mochila é o produto escalar $m \cdot v$.
  
  Dizemos que uma mochila é \textbf{ótima} se ela tem valor máximo.
  
\end{slide}

\begin{slide}
  \stitle{Subestrutura ótima}
  Suponha que $m[1..n]$ é uma mochila ótima para o problema $(w, v, n, W)$.
  
  Se $m[n] = 1$
  
  \vspace{-1cm} \hspace{1cm} então $m[1.. n-1]$ é uma mochila booleana 
  
  \vspace{-1cm} \hspace{2cm}  ótima para o subproblema $(w, v, n-1, W-w[n])$

  \vspace{-1cm} \hspace{1cm} senão $m[1.. n]$ é uma mochila booleana 

  \vspace{-1cm} \hspace{2cm}  ótima para o subproblema $(w, v, n-1, W)$
\end{slide}

\begin{slide}
  \stitle{Simplificação}
  
  \textbf{\darkgreen Problema:}  Encontrar o \textbf{valor} de uma mochila boolean ótima.
  
  \hspace{1.5cm} $t[i, Y] = $ valor de uma mochila booleana ótima 
  
    \vspace{-1cm} \hspace{4.7cm}   para $(w,v,i,W)$
    
 Possíveis valores de $Y = 0,1,2, \cdots , W$    
\end{slide}

\begin{slide}
  \stitle{Recorrência}

  \hspace{1cm} $t[i, Y] = $ valor da expressão $m \cdot v$ 

    \vspace{-1cm} \hspace{4.3cm} sujeito à restrição $m \cdot w \leq Y$ (pesquisa operacional)
    
  \hspace{1cm} $t[0, Y] = 0$ para todo $Y$ 

  \vspace{-1cm}  \hspace{1cm} $t[i, 0] = 0$ para todo $i$
  
  \hspace{1cm} $t[i, Y] = t[i-1, Y]$ se $w[i] > Y$ 
   
  \vspace{-1cm}  \hspace{1cm} $t[i, Y] = \mathrm{max}\{t[i-1, Y], t[i-1, Y - w[i]] + v[i]\}$ se $w[i] \leq Y$
\end{slide}

\begin{slide}
  \stitle{Solução recursiva}
  Devolve o \textbf{valor} de uma mochila booleana ótima para $(w,v,n,W)$.
  
\begin{codebox}
\Procname{$\proc{mochila-recursiva}(w, v, n, W)$}
\li  \If $n = 0$ {\blue \textbf{ou}} $W=0$
\li \Then \Return $0$	
\End
\li  \If $w[n] > W$
\li \Then \Return $\proc{mochila-recursiva}(w, v, n-1, W)$	
\End
\li $a = \proc{mochila-recursiva}(w, v, n-1, W)$	
\li $b = \proc{mochila-recursiva}(w, v, n-1, W-w[n]) + v[n]$	
\li \Return $\mathrm{max}\{a, b\}$
\End
\End
\end{codebox}  
\end{slide}

\begin{slide}
  \stitle{Solução recursiva}
Consumo de tempo no pior caso é $\Omega(2^n)$, pois o mesmo subproblema é resolvido muitas vezes.

Com programação dinâmica cada subproblema, valor de uma mochila booleana ótima para $(w, v, i, Y)$, é resolvido \textbf{apenas uma vez}.

Como é o algoritmo?
\end{slide}

\begin{slide}
  \stitle{Problema da mochila booleana (PD)}
    Devolve o \textbf{valor} de uma mochila booleana ótima para $(w,v,n,W)$.

\vspace{-1cm}\begin{codebox}
\Procname{$\proc{mochila-pd}(w, v, n, W)$}
\li \For $i = 0$ \To $n$ \textbf{\blue faça} $t[i,0] = 0$
\li \For $Y = 0$ \To $W$ \textbf{\blue faça} $t[0,Y] = 0$
\li \For $i = 1$ \To $n$ \textbf{\blue faça}
\li \>  \For $Y = 1$ \To $W$ \textbf{\blue faça}
\li \> \> $a = t[i-1, Y]$
\li \> \> \If $w[i]>Y$ \textbf{\blue então} $b = 0$
\li \> \> \hspace{4.3cm}\textbf{\blue senão} $b = t[i-1, Y-w[i]] + v[i]$
\li \> \> $t[i, Y] = \mathrm{max}\{a, b\}$
\li \Return $t[n,W]$
\End
\end{codebox}   
    
\end{slide}

\begin{slide}
  \stitle{Obtenção da mochila booleana (PD)}

\vspace{-1cm}\begin{codebox}
\Procname{$\proc{mochila}(w, v, n, W)$}
\li $Y = W$
\li \For $i = n$ \To $1$ \textbf{\blue faça}
\li \> \If $(t[i,Y] = t[i-1, Y])$ \textbf{\blue então} $m[i] = 0$
\li \> \> \hspace{6.5cm}\textbf{\blue senão} $m[i] = 1$
\li \> $Y = Y - w[i]$
\li \Return $m$
\end{codebox}   

Complexidade: $O(n)$    
\end{slide}
