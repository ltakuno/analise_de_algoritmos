\begin{slide}
  \stitle{Análise amortizada} 
Útil para analisar sequências de operações ou iterações, que normalmente envolve estruturas de dados.

Serve para melhorar análises de pior caso que baseiam-se no pior caso de uma operação ou iteração.

Diferente de análise de caso médio.
\end{slide}

\begin{slide}
  \stitle{Técnicas de análise amortizada} 
  
  \itemtrig\ Agregado
  
  \vspace{-1cm}\itemtrig\  Contábil
  
  \vspace{-1cm}\itemtrig\  Potencial
  
  Lembre-se que análise amortizada não envolve probabilidade!   
\end{slide}

\begin{slide}
  \stitle{Método agregado} 
  \itemtrig\ $T(n)$ é o custo de uma sequência de $n$ operações.
  
  \vspace{-1cm}\itemtrig\ Custo amortizado é dado por $\displaystyle \frac{T(n)}{n}$.
  
  \vspace{-1cm}\itemtrig\ Cada operação recebe o mesmo custo amortizado, mesmo se são diferentes.
\end{slide}

\begin{slide}
  \stitle{Exemplo da pilha} 
  Considere uma pilha com operações usuais 
  
  \hspace{1cm}$\proc{push}(x, S)$ = empilha x em S

  \vspace{-1cm}\hspace{1cm}$\proc{pop}(S)$ = desempilha do topo de S
  
  Cada operação executa em $O(1)$
  
  O custo de uma sequência de $n$ operações $\proc{push}(x, S)$ e $\proc{pop}(S)$ é $T(n) = n$
\end{slide}

\begin{slide}
Consideremos uma operação $\proc{multipop}(k, S)$

\begin{codebox}
\Procname{$\proc{multipop}(k,S)$}
\li  \While (!$\proc{pilha\_vazia}(S)$ {\blue e} $k \neq 0$) {\blue \textbf{faça}}
\li  \>  $\proc{pop}(S)$
\li  \>    $k = k - 1$
\End
\end{codebox}  

Qual é o tempo de execução do $\proc{multipop()}$?
\end{slide}

\begin{slide}
\itemtrig\ Número de iterações do $\proc{multipop()}$ é $min\{k, |S|\}$

\itemtrig\ Como o custo do $\proc{pop}(S)$ é $1$, então o custo total do $\proc{multipop()}$ é $min\{k, |S|\}$.

\itemtrig\ Portanto, o tempo de execução é $O(min\{k, |S|\})$.
\end{slide}

$$\underbrace{\proc{push} \quad \proc{push} \quad \proc{multipop} \quad \cdots  \quad \proc{pop} \quad \proc{pop} \quad \proc{multipop}}_{n}$$

 
\itemtrig\ Consumo de tempo?  
\end{slide}

\begin{slide}
  \stitle{Análise ingênua}  
\itemtrig\  Considere uma sequência de $n$ operações $\proc{push}$, $\proc{pop}$ e $\proc{multipop}$ em uma pilha inicialmente vazia.

$$\underbrace{\proc{push} \quad \proc{push} \quad \proc{multipop} \quad \cdots  \quad \proc{pop} \quad \proc{pop} \quad \proc{multipop}}_{n}$$

 
\itemtrig\ Custo do pior caso $\proc{push}$ e $\proc{pop}$ é $n$.
\end{slide}

\begin{slide}
  \stitle{Análise ingênua}  
\itemtrig\  Considere uma sequência de $n$ operações $\proc{push}$, $\proc{pop}$ e $\proc{multipop}$ em uma pilha inicialmente vazia.

$$\underbrace{\proc{push} \quad \proc{push} \quad \proc{multipop} \quad \cdots  \quad \proc{pop} \quad \proc{pop} \quad \proc{multipop}}_{n}$$

 
\itemtrig\ Custo do pior caso $\proc{multipop}$ é $n$, pois a pilha tem no máximo $n$ elementos.
\end{slide}

\begin{slide}
  \stitle{Análise ingênua}  
\itemtrig\  Considere uma sequência de $n$ operações $\proc{push}$, $\proc{pop}$ e $\proc{multipop}$ em uma pilha inicialmente vazia.

$$\underbrace{\proc{push} \quad \proc{push} \quad \proc{multipop} \quad \cdots  \quad \proc{pop} \quad \proc{pop} \quad \proc{multipop}}_{n}$$

 
\itemtrig\ Podemos ter $n$ operações $\proc{multipop}$'s na sequência, então o custo total é $O(n^2)$.
\end{slide}

\begin{slide}
  \stitle{Aplicando o método agregado}  
\end{slide}

\begin{slide}
  \stitle{Aplicando o método agregado}  
Embora uma operação $\proc{multipop}$ possa ser ``cara'', qualquer sequência de $n$ operações $\proc{push}$, $\proc{pop}$ e $\proc{multipop}$ executa no máximo em $O(n)$.

\hspace{2cm}\itemtrig\ $n$ $\proc{push}$'s $\Rightarrow$ $n$ $\proc{pop}$'s

\hspace{2cm}\itemtrig\ Custo total é $2n = O(n)$

\hspace{2cm}\itemtrig\ Custo amortizado para cada operação da sequência é: 
$$\displaystyle \frac{2n}{n} = 2 = O(1).$$
\end{slide}

\begin{slide}
  \stitle{Método contábil}  
  \itemtrig\ Atribui cobranças diferentes à operações diferentes.
  
  \itemtrig\ Custo amortizado igual a preço cobrado.
  
  \itemtrig\ Se custo amortizado maior que custo atual, então armazena a diferença como crédito.
  
  \itemtrig\ Use o crédito depois para pagar por operações cujo custo atual é maior que o custo amortizado.
\end{slide}

\begin{slide}
  \stitle{Método contábil}  
  
 \itemtrig\ $c_i$ é o custo atual da $i-$ésima operação.
 
 \itemtrig\ $\hat{c}_i$ é o custo amortizado da $i-$ésima operação.
 
 \itemtrig\ Então, $\displaystyle \sum^n_{i=1} \hat{c}_i \geq \sum^n_{i=1} c_i$  
 
 \itemtrig\ Crédito armazenado:  $\displaystyle \sum^n_{i=1} \hat{c}_i - \sum^n_{i=1} c_i \geq 0$
 \end{slide}

\begin{slide}
  \stitle{Método contábil} 
  \begin{center}
  \begin{tabular}{ccc} \hline
  operação      & custo atual & custo amortizado \\ \hline
  $\proc{push}$ &  1          & 2  \\
  $\proc{pop}$ &  1          & 0  \\
  $\proc{multipop}$ &  $min\{k, |S|\}$          & 0  \\ \hline
  \end{tabular}
  \end{center}

  Intuição envolvida: Cobra $R\$ 2,00$ para empilhar.

\hspace{1cm}\itemtrig\  $R\$ 1,00$ para o $\proc{push}$

\vspace{-1cm}\hspace{1cm}\itemtrig\  $R\$ 1,00$ é pré-pagamento para desempilhar com $\proc{pop}$ ou $\proc{multipop}$.
\end{slide}

\begin{slide}
  \stitle{Método contábil}  
Como cada objeto tem  $R\$ 1,00$, que é crédito, o crédito armazenado nunca fica negativo.

Portanto, para $n$ operações o custo é $2n = O(n)$.
\end{slide}

\begin{slide}
  \stitle{Método potencial}  
\itemtrig\ Estrutura inicial $D_0$, onde $n$ operações são realizadas, resultando em uma sequência $D_0, D_1, D_2, \cdots D_n$.

\itemtrig\ $c_i$ é o custo real (atual) da $i$-ésima operação.

\itemtrig\ Uma função potencial $\Upphi$, atribui a cada estrutura $D_i$ um número real $\Upphi(D_i)$.
\end{slide}

\begin{slide}
  \stitle{Método potencial}  
O custo amortizado da $i$-ésima operação é definida por:

\vspace{-1cm}$$\hat{c}_i = c_i + \underbrace{\Upphi(D_i) - \Upphi(D_{i-1})}_{\textrm{''energia potencial´´}}$$
 
\vspace{-1cm} O custo amortizado total é:

\vspace{-1cm}$$\displaystyle \sum^n_{i=1}\hat{c}_i = \left(\sum^n_{i=1} c_i\right) + \Upphi(D_i) - \Upphi(D_{i-1})$$

Se $\Upphi(D_n) - \Upphi(D_0) \geq 0$, então $\displaystyle \sum^n_{i=1}\hat{c}_i$ é um limitante superior para o custo real (atual).
\end{slide}

\begin{slide}
  \stitle{Método potencial} 
Exemplo da pilha:

\itemtrig\ $\Upphi(D_i)$ = número de elementos na pilha ($|D_i|$)

\vspace{-0.7cm}\itemtrig\ $\Upphi(D_0)$ = 0 (início)
  
\vspace{-0.7cm}\itemtrig\ $\Upphi(D_n)$ = $|D_n|$ (final)

\vspace{-0.7cm}\itemtrig\ Note que $\Upphi(D_n) - \Upphi(D_0) \geq 0$

Quando uma operação $\proc{push}$ é executada em uma pilha com $|D_{i-1}| = s$ elementos, a variação de energia potencial é:

\vspace{-1.5cm}$$\Upphi(D_i) - \Upphi(D_{i-1}) = (s+1)-s = 1$$
\end{slide}

\begin{slide}
  \stitle{Método potencial} 
  Lembremos que o custo real (atual) da operação é $c_i=1$  
  
  Custo amortizado da operação $\proc{push}$:

$$\hat{c}_i = c_i + \underbrace{\Upphi(D_i) - \Upphi(D_{i-1})}_{1} = 1 + 1 = 2$$  
  
  Custo amortizado da operação $\proc{pop}$: $0$ (zero)
\end{slide}

\begin{slide}
  \stitle{Método potencial} 
  Suponha que a $i$-ésima operação executa $\proc{multipop}(k, S)$ e assim $\bar{K} = min\{k , |S|\}$ são desempilhados.
  
  Custo real(atual) da operação é $c_i = \bar{K}$
  
  A variação de energia potencial é:

\vspace{-1.5cm}$$\Upphi(D_i) - \Upphi(D_{i-1}) = (|S|-\bar{K})-|S| = -\bar{K}$$

Custo amortizado:

\vspace{-1.5cm}$$\hat{c}_i =  c_i + \Upphi(D_i) - \Upphi(D_{i-1}) = \bar{K} - \bar{K}$$
\end{slide}

\begin{slide}
  \stitle{Método potencial} 
  O custo amortizado total de $n$ operações é $O(n)$.
  
  Portanto, o custo real de $n$ operações no pior caso é $O(n)$.
  
  Lembre-se que o custo amortizado é um limite superior de custo real.  
\end{slide}
