\begin{slide}
  \stitle{Notação ``Chão'' e ``Teto''}
	
\begin{definition}
O \textbf{chão} (ou piso) de um número não negativo $x$ é o único natural $i$ tal que $i \leq x < i+1$.
O chão de $x$ é denotado por $\lfloor x \rfloor$.
\end{definition}	


\begin{definition}
O \textbf{teto} de $x$ é o único número $j$ natural tal que que $j-1 < x \leq j$. O teto de $x$ é denotado por $\lceil x \rceil$
\end{definition}	

Propriedade: $\lceil \frac{x}{2} \rceil + \lfloor \frac{x}{2} \rfloor = x$

\end{slide}

\begin{slide}
  \stitle{Métodos para resolver a recorrência}
  \begin{itemize}
  \item \textbf{Substituição}: Arriscamos um ``palpite'' para um limite e então usamos indução matemática para provar que nosso ``palpite'' estava certo.
  \item \textbf{Árvore de recursão}: Convertemos recorrência em uma árvore e usamos a técnica para limitar somatórios com intuito de resolver a recorrência.
  
  \item \textbf{Método mestre}: Para recorrências da forma $T(n) = aT(n/b) + f(n)$, com $a \geq 1$ e $b > 1$.
  
  \end{itemize}
  
\end{slide}

\begin{slide}
  \stitle{Lembre-se}
  
  A meta não é determinar as soluções exatas de uma recorrência, mas encontrar uma solução $g(n)$ tal que $T(n) = \Theta(g(n))$ 
\end{slide}

\begin{slide}
  \stitle{Método Mestre}
  
  Recorrências de algoritmos que requerem chamadas recursivas para resolver problemas de tamanho $(n/b)$ com custo $f(n)$ em passos de divisão e conquista podem ser escritas (com constantes $a\geq 1$ e $b>1$) da seguinte forma:
  
  $$T(n) = aT(n/b) + f(n)$$
 \end{slide}

\begin{slide}
  \stitle{Método Mestre} 
  
    
Assumindo que a base indutiva tem custo constante, temos um método geral para descrever solução de algumas destas recorrências. 

A recorrência anterior também inclui casos $\lceil n/b \rceil$ e $\lfloor n/b \rfloor$.
\end{slide}

\begin{slide}
  \stitle{Teorema Mestre} 
  
Sejam $a \geq 1$ e $b > 1$ constantes, $f(n)$ uma função e $T(n)$ definida para inteiros não-negativos como a recorrência.

$$T(n) = aT(n/b) + f(n)$$
\end{slide}

\begin{slide}
  \stitle{Teorema Mestre} 
  
Então, $T(n)$ tem os seguintes limites assintóticos

\begin{enumerate}
\item[(1)] Se $f(n) = O(n^{\log_b a - \varepsilon})$ para alguma constante $\varepsilon > 0$, então $T(n) = \Theta(n^{\log_b a})$
\item[(2)] Se $f(n) = \Theta(n^{\log_b a})$, então $T(n) = \Theta(n^{\log_b a} \cdot \log n)$
\item[(3)] Se $f(n) = \Omega(n^{\log_b a + \varepsilon})$ para alguma constante $\varepsilon > 0$, e se $af(n/b) \leq c f(n)$ para alguma constante $c < 1$ e todos $n$ suficientemente grandes, então $T(n) = \Theta(f(n))$
\end{enumerate}
\end{slide}
