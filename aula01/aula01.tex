\begin{slide}
  \stitle{Ordenação por inserção}
\begin{algorithmic}[1]
  \TITLE{\textsc{Insertion-Sort}$(A,n)$}
  \FOR{$j=2$ {\blue até} $n$}
  \STATE $\mathit{chave} = A[j]$
  \STATE \COMMENT{Insere $A[j]$ na sequência ordenada $A[1..j-1]$}
  \STATE $i = j-1$
  \WHILE{$i>0$ {\blue e} $A[i]>chave$}
  \STATE $A[i+1]= A[i]$
  \STATE $i = i-1$
  \ENDWHILE
  \STATE $A[i+1] = \mathit{chave}$
  \ENDFOR
\end{algorithmic}
  
\end{slide}

\begin{slide}
  \stitle{Ordenação por inserção}
\begin{algorithmic}[1]
  \TITLE{\textsc{Insertion-Sort}$(A,n)$}
    \algcost{\textit{cost} }{ \textit{times}}
  \FOR{$j=2$ {\blue até} $n$
    \algcost{$c_1$}{$n$}}
  \STATE $\mathit{chave} = A[j]$
    \algcost{ $c_2$}{ $n-1$}
  \STATE \COMMENT{Insere $A[j]$ na sequência ordenada $A[1..j-1]$}
    \algcost{ $0$}{ $n-1$}
  \STATE $i = j-1$
    \algcost{ $c_4$}{ $n-1$}
  \WHILE{$i>0$ {\blue e} $A[i]>chave$
    \algcost{$c_5$}{$\sum_{j=2}^{n} t_j$}}
  \STATE $A[i+1]= A[i]$
    \algcost{ $c_6$}{ $\sum_{j=2}^{n} (t_j-1)$}
  \STATE $i = i-1$
    \algcost{$c_7$}{$\sum_{j=2}^{n} (t_j-1)$}
  \ENDWHILE
  \STATE $A[i+1] = \mathit{chave}$
    \algcost{$c_8$}{$n-1$}
  \ENDFOR
\end{algorithmic}
$t_j$ é o número de vezes que o teste do while da linha 5 é executado para cada $j$.
\end{slide}

\begin{slide}
  \stitle{Notação $\Theta$}
  
  \begin{definition}
  Para uma dada função $g(n)$, denotamos por $\theta(g(n))$ o conjunto de funções:
  
  $$\begin{array}{ll}\Theta(g(n)) = &\{ f(n) : \exists c1, c2 > 0 \textrm{ e }  n_0 > 0 \textrm{ tal que } \\
                                    & c_1 g(n) \leq f(n) \leq c_2 g(n), \forall n \geq n_0     \}\end{array}$$
                                    
  \end{definition}
\end{slide}

\begin{slide}
  \stitle{Notação $\Theta$}  
  Em vez de escrever $f(n) \in \Theta(g(n))$ escreveremos $f(n) = \Theta(g(n))$
\begin{center}
\psfrag{c1gn}{$c_1 g(n)$}
\psfrag{c2gn}{$c_2 g(n)$}
\psfrag{fn}{$f(n)$}
\psfrag{no}{$n_0$}
\includegraphics[height=8.0cm]{fig01.eps}
\end{center} 

A notação $\Theta$ é usada para delimitações exatas de funções.
\end{slide}

\begin{slide}
  \stitle{Exemplo: Notação $\Theta$}  

$$\begin{array}{l}
\displaystyle \frac{1}{2} n^2 - 3n = \Theta(n^2) \\\\
\displaystyle  \frac{1}{2} n^2 - 3n \leq \frac{1}{2} n^2, \forall n \geq 0 \rightarrow c_2 = \frac{1}{2}, n_0^{(2)} \geq 0.\\\\
\displaystyle  \frac{1}{4} n^2 \leq \frac{1}{2} n^2 - 3n \rightarrow c_1 = \frac{1}{4}, n_0^{(1)} \geq 12 \\\\
n_0 = max\{0,12\} = 12
\end{array} 
$$

Portanto, $\displaystyle c_1 = \frac{1}{4}$, $\displaystyle c_2 = \frac{1}{2}$ e $n_0 = 12$
\end{slide}

\begin{slide}

  \begin{definition}
  Para uma dada função $g(n)$, denotamos por $O(g(n))$ o conjunto de funções:
  
  $$\begin{array}{ll}O(g(n)) = &\{ f(n) : \exists c > 0 \textrm{ e }  n_0 > 0 \textrm{ tal que } \\

  \end{definition}
\end{slide}

  \stitle{Exemplos: Notação $O$}  
  
  \begin{itemize}
  \item[(a)] $n = O(n^2)$?

  Sim, tome $c = 1$ e $n_0 = 1$
  
  \item[(b)] $\frac{1}{2}n^2 - 3n = O(n^2)$? 
  
  Sim , tome $c = \frac{1}{2}$ e $n_0 = 7$
  \end{itemize}
\end{slide}

\begin{slide}
  \stitle{Notação $\Omega$}  
  \begin{definition}
  Para uma dada função $g(n)$, denotamos por $\Omega(g(n))$ o conjunto de funções:
  
  $$\begin{array}{ll}\Omega(g(n)) = &\{ f(n) : \exists c > 0 \textrm{ e }  n_0 > 0 \textrm{ tal que } \\
                                    &  f(n) \geq c g(n), \forall n \geq n_0     \}\end{array}$$
  
  \end{definition}
\end{slide}

\begin{slide}
  \stitle{Exemplos: Notação $\Omega$}  
 A notação $O$ serve para atribuir delimitações inferiores para a função $g(n)$.

  \begin{itemize}
  \item[(a)] $\frac{1}{3}n^2 - 3n  = \Omega(n^2)$?

  Sim, tome $c = \frac{1}{4}$ e $n_0 = 7$
  
  \item[(b)] $n^2 - 2n = \Omega(n)$? 
  
  Sim , tome $c = 10$ e $n_0 = 8$
  \end{itemize}

\end{slide}

\begin{slide}
  \stitle{Notação $o$ (oh pequeno)}    
  Para quando há apenas um \textbf{limite assintótico superior}, sem permitir $f(n) = cg(n)$.
  
  Utiliza-se a notação $o$ para denotar um limitante superior que é assintoticamente restrito.
\end{slide}

\begin{slide}
  \stitle{Notação $o$ (oh pequeno)}   
  \begin{definition}
  Para uma dada função $g(n)$, denotamos por $o(g(n))$ o conjunto de funções:
  
  $$\begin{array}{ll}o(g(n)) = &\{ f(n) : \textrm{ para qualquer constante positiva } c \textrm{ existe } \\  & \textrm{uma constante positiva }  n_0  \textrm{ tal que } 
                                  f(n) < c g(n), \forall n \geq n_0     \}\end{array}$$
  
  \end{definition}
  
  Intuitivamente, $f$ é dominado por $g$ assintoticamente.
\end{slide}

\begin{slide}
  \stitle{Notação $o$ (oh pequeno)}   
  Por exemplo, $2n = o(n^2)$, mas $2 n^2 \neq o(n^2)$
  
  Intuitivamente, na notação $o$, a função $f(n)$ torna-se insignificante em relação a $g(n)$ quando vai para o infinito. Ou seja,
  
  $$\displaystyle \lim_{n \rightarrow \infty} \frac{f(n)}{g(n)} = 0$$
\end{slide}

\begin{slide}
  \stitle{Notação $o$ (oh pequeno)}
    Exemplo:  $2n = o(n^2)$
    
  $$\displaystyle \lim_{n \rightarrow \infty} \frac{2n}{n ^ 2} = \lim_{n \rightarrow \infty} \frac{2}{n} = 0$$
 
   Portanto, $2n = o(n^2)$.
\end{slide}

  \stitle{Notação $\omega$ (omega pequeno)}   
\end{slide}

\begin{slide}
  \stitle{Notação $\omega$ (omega pequeno)}   
  Por analogia, a notação $\omega$ está para a notação $\Omega$, assim como a notação $o$ está para a notação $O$. 

  Então, para quando há apenas um limite assintótico inferior, sem permitir $f(n) = cg(n)$.  
  
  Utiliza-se a notação $\omega$ para denotar um limitante inferior que não é assintoticamente restrito.
\end{slide}

\begin{slide}
  \stitle{Notação $\omega$ (omega pequeno)}   
  \begin{definition}
  Para uma dada função $g(n)$, denotamos por $\omega(g(n))$ o conjunto de funções:
  
  $$\begin{array}{ll}o(g(n)) = &\{ f(n) : \textrm{ para qualquer constante positiva } c \textrm{ existe } \\  & \textrm{uma constante positiva }  n_0  \textrm{ tal que } 
                                  c g(n) < f(n)  , \forall n \geq n_0     \}\end{array}$$
  
  \end{definition}
  
  Intuitivamente, $f$ domina $g$ assintoticamente.
\end{slide}


\begin{slide}
  \stitle{Notação $\omega$ (omega pequeno)}   
    Por exemplo, $2n^2 = \omega(n)$, mas $2n^2 \neq \omega(n^2)$
    
    Intuitivamente na notação $\omega(n)$ a função $f(n)$ domina $cg(n)$ quando $n$ vai para o infinito. Ou seja, $f(n) = \omega(g(n))$ indica que 
    
$$\displaystyle \lim_{n \rightarrow \infty} \frac{f(n)}{g(n)} = \infty$$    
    
\end{slide}

\begin{slide}
  \stitle{Notação $\omega$ (omega pequeno)}   
    Por exemplo, $n^2/2 = \omega(n)$.
    
  $$\displaystyle \lim_{n \rightarrow \infty} \frac{n^2/2}{n} = \lim_{n \rightarrow \infty} \frac{n}{2} = \infty$$
 

   Portanto, $n^2/2 = \omega(n)$.
\end{slide}

\begin{slide}
  \stitle{Comparação de funções}      
  \itemtrig\ Transitividade
  
  \hspace{1cm} $f(n) = \Theta(g(n))$ e $g(n) = \Theta(h(n))$ então $f(n) = \Theta(h(n))$
  
 \vspace{-1cm} \hspace{1cm} $f(n) = O(g(n))$ e $g(n) = O(h(n))$ então $f(n) = O(h(n))$

 \vspace{-1cm} \hspace{1cm} $f(n) = \Omega(g(n))$ e $g(n) = \Omega(h(n))$ então $f(n) = \Omega(h(n))$

 \vspace{-1cm}  \hspace{1cm} $f(n) = o(g(n))$ e $g(n) = o(h(n))$ então $f(n) = o(h(n))$

 \vspace{-1cm}  \hspace{1cm} $f(n) = \omega(g(n))$ e $g(n) = \omega(h(n))$ então $f(n) = \omega(h(n))$
\end{slide}

\begin{slide}
  \stitle{Comparação de funções}      
  \itemtrig\ Reflexividade
  
    \hspace{1cm} $f(n) = \Theta(f(n))$

    \vspace{-1cm} \hspace{1cm} $f(n) = O(f(n))$

    \vspace{-1cm} \hspace{1cm} $f(n) = \Omega(f(n))$

    \vspace{-1cm} \hspace{1cm} $f(n) \neq o(f(n))$

    \vspace{-1cm} \hspace{1cm} $f(n) \neq \omega(f(n))$
\end{slide}

\begin{slide}
  \stitle{Comparação de funções}      
  \itemtrig\ Simetria
  
    $f(n) = \Theta(g(n))$, se somente se, $g(n) = \Theta(f(n))$

  \itemtrig\ Simetria transposta

    $f(n) = O(g(n))$, se somente se, $g(n) = \Omega(f(n))$

 \vspace{-1cm}  $f(n) = o(g(n))$, se somente se, $g(n) = \omega(f(n))$
\end{slide}

\begin{slide}
  \stitle{Analogia entre notação assintótica e números reais}        
  Sejam $f$, $g$ funções e $a, b \in R$
  
    \hspace{1cm} $f(n) = O(f(n)) \approx a\leq b $

    \vspace{-1cm} \hspace{1cm} $f(n) = \Omega(f(n)) \approx a \geq b$
    
    \vspace{-1cm} \hspace{1cm} $f(n) = \Theta(f(n)) \approx a = b$

    \vspace{-1cm} \hspace{1cm} $f(n) = o(f(n)) \approx a < b$

    \vspace{-1cm} \hspace{1cm} $f(n) = \omega(f(n)) \approx a > b$
\end{slide}

\begin{slide}
  \stitle{Tricotomia}   
  Para quaisquer dois números reais a e b, exatamente uma das sequintes sentenças deve ser verdadeira: 
  $a < b$, $a = b$ ou $a > b$.
  
  Apesar de quaisquer dois números poderem ser comparados, nem todas as funções são assintoticamente comparáveis. Ou seja, para duas funções $f(n)$ e $g(n)$, pode ser que nem $f(n) = O(g(n))$ nem $f(n) = \Omega(g(n))$.
  
\end{slide}

\begin{slide}
  \stitle{Tricotomia}   
  
Exemplo:  Considere as funções $f(n) = n$ e $g(n) = n^{1 + \sin n}$

Neste exemplo, a expressão $1 + \sin n$ varia entre $0$ e $2$.

$$
n^{1 + \sin n} = \left \{ \begin{array}{ll}
1 & \textrm{se } n \in \{ 3\pi/2, 7\pi/2, \cdots \} \\
n & \textrm{se } n \in \{ \pi, 3\pi, \cdots \} \\
n^2 & \textrm{se } n \in \{ \pi/2, 5\pi/2, \cdots \} \\
\end{array}  \right.
$$

Não se pode afirmar que $f(n) = O(g(n))$ e nem $f(n) = \Omega(g(n))$.
\end{slide}

\begin{slide}
  \stitle{Notação assintótica em expressões}
  Como interpretar, por exemplo, $2n^2+n-1 = 2n^2+\Theta(n)$ ?  
  
  Existem uma função $f(n) = \Theta(n)$ tal que 
  
  $$2n^2+n-1 = 2n^2+f(n).$$
  
  Neste caso, $f(n) = n - 1$.
\end{slide}

\begin{slide}
  \stitle{Notação assintótica em expressões}
  E a expressão $2n^2 + \Theta(n) = \Theta(n^2)$ ?
  
  Para toda função $f(n) = \Theta(n)$, existe uma função $g(n) = \Theta(n^2)$ tal que $2n^2 + f(n) = g(n)$.
  
  Por exemplo, 
  
  $f(n) =n - \log n + 5 = \Theta(n)$ e 
  
  $g(n) = 2n^2 + n - \log n + 5 = \Theta(n^2)$.
  \end{slide}
